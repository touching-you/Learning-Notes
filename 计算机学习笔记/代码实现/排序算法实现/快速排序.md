    left = start+1
    right = end
    while(left<right):
        while(left<right and arr[left]<pivot):
            left+=1

            arr[left],arr[right]=swap(arr,left,right)
            right-=1
    if(left == right and arr[right]>pivot):

    if(right!=start):
        arr[start],arr[right]=swap(arr,start,right)
    return (right,arr)
